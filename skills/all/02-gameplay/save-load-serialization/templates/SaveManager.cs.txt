using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using UnityEngine;

namespace {{NAMESPACE}}.SaveSystem
{
    /// <summary>
    /// Central manager for saving and loading game state.
    /// Collects all ISaveable objects and persists their state.
    /// </summary>
    public class SaveManager : MonoBehaviour
    {
        [Header("Configuration")]
        [SerializeField] private string _saveFileName = "savegame";
        [SerializeField] private string _saveFileExtension = ".json";
        [SerializeField] private bool _useEncryption = false;
        
        private static SaveManager _instance;
        public static SaveManager Instance => _instance;
        
        public event Action OnSaveStarted;
        public event Action OnSaveCompleted;
        public event Action OnLoadStarted;
        public event Action OnLoadCompleted;
        public event Action<string> OnError;

        private void Awake()
        {
            if (_instance != null && _instance != this)
            {
                Destroy(gameObject);
                return;
            }
            _instance = this;
            DontDestroyOnLoad(gameObject);
        }

        /// <summary>
        /// Save game to default slot.
        /// </summary>
        public void Save() => SaveToSlot(0);

        /// <summary>
        /// Save game to specific slot.
        /// </summary>
        public void SaveToSlot(int slotIndex)
        {
            OnSaveStarted?.Invoke();
            
            try
            {
                var saveData = new SaveData
                {
                    Timestamp = DateTime.UtcNow.ToString("o"),
                    Version = Application.version,
                    SaveStates = new Dictionary<string, object>()
                };
                
                // Collect all saveables
                var saveables = FindAllSaveables();
                foreach (var saveable in saveables)
                {
                    try
                    {
                        saveData.SaveStates[saveable.SaveId] = saveable.CaptureState();
                    }
                    catch (Exception ex)
                    {
                        Debug.LogError($"[SaveManager] Failed to capture {saveable.SaveId}: {ex.Message}");
                    }
                }
                
                // Serialize
                string json = JsonUtility.ToJson(new SaveDataWrapper(saveData), true);
                
                if (_useEncryption)
                    json = Encrypt(json);
                
                // Write to file
                string path = GetSavePath(slotIndex);
                File.WriteAllText(path, json);
                
                Debug.Log($"[SaveManager] Saved to {path}");
                OnSaveCompleted?.Invoke();
            }
            catch (Exception ex)
            {
                Debug.LogError($"[SaveManager] Save failed: {ex.Message}");
                OnError?.Invoke(ex.Message);
            }
        }

        /// <summary>
        /// Load game from default slot.
        /// </summary>
        public void Load() => LoadFromSlot(0);

        /// <summary>
        /// Load game from specific slot.
        /// </summary>
        public void LoadFromSlot(int slotIndex)
        {
            OnLoadStarted?.Invoke();
            
            try
            {
                string path = GetSavePath(slotIndex);
                
                if (!File.Exists(path))
                {
                    Debug.LogWarning($"[SaveManager] No save file at {path}");
                    return;
                }
                
                string json = File.ReadAllText(path);
                
                if (_useEncryption)
                    json = Decrypt(json);
                
                var wrapper = JsonUtility.FromJson<SaveDataWrapper>(json);
                var saveData = wrapper.ToSaveData();
                
                // Restore all saveables in priority order
                var saveables = FindAllSaveables()
                    .OrderBy(s => s.LoadPriority)
                    .ToList();
                
                foreach (var saveable in saveables)
                {
                    if (saveData.SaveStates.TryGetValue(saveable.SaveId, out var state))
                    {
                        try
                        {
                            saveable.RestoreState(state);
                        }
                        catch (Exception ex)
                        {
                            Debug.LogError($"[SaveManager] Failed to restore {saveable.SaveId}: {ex.Message}");
                        }
                    }
                }
                
                Debug.Log($"[SaveManager] Loaded from {path}");
                OnLoadCompleted?.Invoke();
            }
            catch (Exception ex)
            {
                Debug.LogError($"[SaveManager] Load failed: {ex.Message}");
                OnError?.Invoke(ex.Message);
            }
        }

        /// <summary>
        /// Check if a save slot exists.
        /// </summary>
        public bool HasSave(int slotIndex = 0)
        {
            return File.Exists(GetSavePath(slotIndex));
        }

        /// <summary>
        /// Delete a save slot.
        /// </summary>
        public void DeleteSave(int slotIndex = 0)
        {
            string path = GetSavePath(slotIndex);
            if (File.Exists(path))
                File.Delete(path);
        }

        private List<ISaveable> FindAllSaveables()
        {
            return FindObjectsByType<MonoBehaviour>(FindObjectsSortMode.None)
                .OfType<ISaveable>()
                .ToList();
        }

        private string GetSavePath(int slotIndex)
        {
            string fileName = slotIndex == 0 
                ? $"{_saveFileName}{_saveFileExtension}"
                : $"{_saveFileName}_{slotIndex}{_saveFileExtension}";
            return Path.Combine(Application.persistentDataPath, fileName);
        }

        private string Encrypt(string data)
        {
            // Simple XOR encryption - replace with proper encryption in production
            return Convert.ToBase64String(
                System.Text.Encoding.UTF8.GetBytes(data)
            );
        }

        private string Decrypt(string data)
        {
            return System.Text.Encoding.UTF8.GetString(
                Convert.FromBase64String(data)
            );
        }
    }

    [System.Serializable]
    public class SaveData
    {
        public string Timestamp;
        public string Version;
        public Dictionary<string, object> SaveStates;
    }

    // Wrapper for Unity's JsonUtility (doesn't support Dictionary)
    [System.Serializable]
    public class SaveDataWrapper
    {
        public string Timestamp;
        public string Version;
        public List<SaveEntry> Entries = new();

        public SaveDataWrapper() { }
        
        public SaveDataWrapper(SaveData data)
        {
            Timestamp = data.Timestamp;
            Version = data.Version;
            foreach (var kvp in data.SaveStates)
            {
                Entries.Add(new SaveEntry 
                { 
                    Key = kvp.Key, 
                    Value = JsonUtility.ToJson(kvp.Value) 
                });
            }
        }

        public SaveData ToSaveData()
        {
            var data = new SaveData
            {
                Timestamp = Timestamp,
                Version = Version,
                SaveStates = new Dictionary<string, object>()
            };
            foreach (var entry in Entries)
            {
                data.SaveStates[entry.Key] = entry.Value; // Still JSON string
            }
            return data;
        }
    }

    [System.Serializable]
    public class SaveEntry
    {
        public string Key;
        public string Value;
    }
}
