using UnityEngine;

namespace {NAMESPACE}
{
    public enum EffectType { Buff, Debuff }

    [CreateAssetMenu(fileName = "NewEffect", menuName = "Game/Effects/Status Effect")]
    public class EffectData : ScriptableObject
    {
        public string Name;
        public EffectType Type;
        public Sprite Icon;
        public float Duration = 5f;
        public bool IsStackable;
        
        [Header("Ticking (DOT)")]
        public bool HasTick;
        public float TickRate = 1f;
        public float TickAmount = 5f; // e.g. Damage per tick
    }
}
