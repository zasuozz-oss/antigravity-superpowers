// =============================================================================
// GitWorkflowTools.cs - Git Setup Automation
// Editor tool to generate .gitignore and .gitattributes for Unity
// =============================================================================

#if UNITY_EDITOR
using System.IO;
using UnityEditor;
using UnityEngine;

namespace YourGame.Editor
{
    /// <summary>
    /// Automation tool to setup Git environment for Unity.
    /// </summary>
    public static class GitWorkflowTools
    {
        [MenuItem("Tools/Git/Generate .gitignore")]
        public static void GenerateGitIgnore()
        {
            string path = Path.Combine(Application.dataPath, "../.gitignore");
            if (File.Exists(path) && !EditorUtility.DisplayDialog("Exists", ".gitignore exists. Overwrite?", "Yes", "No"))
                return;

            File.WriteAllText(path, GetGitIgnoreContent());
            Debug.Log("Generated .gitignore at project root.");
        }

        [MenuItem("Tools/Git/Generate .gitattributes (LFS)")]
        public static void GenerateGitAttributes()
        {
            string path = Path.Combine(Application.dataPath, "../.gitattributes");
            if (File.Exists(path) && !EditorUtility.DisplayDialog("Exists", ".gitattributes exists. Overwrite?", "Yes", "No"))
                return;

            File.WriteAllText(path, GetGitAttributesContent());
            Debug.Log("Generated .gitattributes at project root.");
        }

        private static string GetGitIgnoreContent()
        {
            return @"
/[Ll]ibrary/
/[Tt]emp/
/[Oo]bj/
/[Bb]uild/
/[Bb]uilds/
/[Ll]ogs/
/[Uu]ser[Ss]ettings/

# Autogenerated VS/MD/Consulo solution and project files
ExportedObj/
.consulo/
*.csproj
*.unityproj
*.sln
*.suo
*.tmp
*.user
*.userprefs
*.pidb
*.booproj
*.svd
*.pdb
*.opendb
*.VC.db

# Unity3D generated meta files
*.pidb.meta
*.pdb.meta
*.mdb.meta

# OS generated
.DS_Store
.DS_Store?
._*
.Spotlight-V100
.Trashes
ehthumbs.db
Thumbs.db
";
        }

        {
