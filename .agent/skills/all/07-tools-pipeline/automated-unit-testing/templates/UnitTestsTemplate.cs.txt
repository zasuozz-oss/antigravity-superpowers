    /// <summary>
    /// PLAY MODE SPECIFIC
    /// Tests MonoBehaviours, Physics, Coroutines, and Engine features.
    /// Slower, real-time, good for integration testing.
    /// </summary>
    public class PlayModeIntegrationTests
    {
        private GameObject _testObject;

        [SetUp]
        public void Setup()
        {
            // Created before each test
            _testObject = new GameObject("TestObject");
        }

        [TearDown]
        public void Teardown()
        {
            // Cleaned up after each test
            if (_testObject != null)
            {
                Object.Destroy(_testObject);
            }
        }

        [UnityTest]
        public IEnumerator Physics_Gravity_MovesObjectDown()
        {
            // Arrange
            var rb = _testObject.AddComponent<Rigidbody>();
            rb.useGravity = true;
            Vector3 startPos = _testObject.transform.position;

            // Act
            yield return new WaitForSeconds(0.5f); // Simulate physics time

            // Assert
            Assert.Less(_testObject.transform.position.y, startPos.y);
        }
